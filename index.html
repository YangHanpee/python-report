用 Python 計算平方

請輸入一個數字，我會幫你計算它的平方：

結果：

from js import document def calculate_square(event): # 獲取輸入的值 user_input = document.querySelector("#user-input").value if user_input: try: num = float(user_input) square = num ** 2 # 顯示計算結果 document.querySelector("#result").innerText = f"{square:.2f}" except ValueError: document.querySelector("#result").innerText = "請輸入有效的數字！" # 綁定按鈕點擊事件 document.querySelector("#calculate-button").addEventListener("click", calculate_square)
